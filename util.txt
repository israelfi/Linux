ls /dev/char # mostra as portas

lsusb # Mostra o nomes dos dispositivos conectados

cat /proc/bus/input/devices # Mostra o nomes dos dispositivos conectados

sudo jstest /dev/input/js0 # Testa o joystick na porta js0

history | grep (coisa) # Pesquisa (coisa) no histórico do terminal

grep '(coisa)' (arquivo) # Mostra as ocorrencias de (coisa) ni (arquivo)

ps aux # lista todos os processos para o standard output

env # Mostra as variaveis de ambiente

echo $LANG # variavel que especifica a linguagem do usuario (pt_BR para portugues do Brasil)

# PATH é uma variável que contém de ambiente que contém uma lista de diretórios delimitados por :. É nessa lista que o bash procura o programa para executar quando se digita um comando no shell. Caso apareça o erro "command not found" pode ser que exista um erro com o PATH.

# Variáveis da shell são usadas somente pelo shell. $ DRINK="water". $ echo $DRINK -> water. $ OI="have some $DRINK". $ echo $OI -> have some water

# VAraiáveis de ambiente são variáveis de shell que foram exportadas, estando acessíveis para os processos filhos. $ export DRINK="water"

unset VARIAVEL # Retira a variável de ambiente VARIAVEL

export PATH="/path/to/dir:$PATH" # Adiciona /path/to/dir ao PATH
